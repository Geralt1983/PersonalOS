2. The Anchor: Physics-Based Interaction
Update components/the-anchor.tsx. We are replacing standard CSS transitions with framer-motion springs. This gives the toggles "weight" and satisfaction.
"use client"

import { useState } from "react"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { motion } from "framer-motion"

export function TheAnchor() {
  const [anchors, setAnchors] = useState([
    { id: 1, label: "Hydrate (32oz)", active: true },
    { id: 2, label: "No Phone in Bed", active: false },
    { id: 3, label: "Read Physics (10m)", active: false },
  ])

  const toggleAnchor = (id: number) => {
    // Optimistic UI: State updates instantly
    setAnchors(anchors.map(a => a.id === id ? { ...a, active: !a.active } : a))
    
    // Trigger Haptic Feedback (Mobile only)
    if (typeof navigator !== 'undefined' && navigator.vibrate) {
        navigator.vibrate(10);
    }
  }

  return (
    <Card className="border border-white/10 bg-black/40 backdrop-blur-xl shadow-2xl relative overflow-hidden">
      <CardHeader className="pb-4">
        <CardTitle className="text-xs font-bold tracking-[0.2em] text-cyan-500 uppercase flex items-center gap-2">
          The Anchor
        </CardTitle>
      </CardHeader>

      <CardContent className="flex flex-col gap-3">
        {anchors.map((anchor) => (
          <motion.div 
            key={anchor.id} 
            onClick={() => toggleAnchor(anchor.id)}
            className="group cursor-pointer relative"
            whileTap={{ scale: 0.98 }} // Physics: button shrinks slightly on press
          >
            {/* The Switch Container */}
            <div className={`
              relative h-14 rounded-xl border flex items-center px-4 transition-colors duration-500
              ${anchor.active 
                ? 'bg-blue-500/10 border-blue-400/30' 
                : 'bg-white/5 border-white/5 hover:bg-white/10'
              }
            `}>
              
              {/* The "Glow" behind the active state */}
              {anchor.active && (
                 <motion.div 
                    layoutId="glow"
                    className="absolute inset-0 bg-blue-500/10 blur-xl"
                    initial={{ opacity: 0 }}
                    animate={{ opacity: 1 }}
                 />
              )}

              {/* Text */}
              <span className={`flex-1 text-sm font-medium z-10 transition-colors ${anchor.active ? 'text-blue-100' : 'text-gray-400'}`}>
                {anchor.label}
              </span>

              {/* Physical Switch UI */}
              <div className="relative w-12 h-7 rounded-full bg-black/50 border border-white/10 p-1 flex items-center">
                 <motion.div 
                    className={`w-5 h-5 rounded-full shadow-lg ${anchor.active ? 'bg-cyan-400' : 'bg-gray-600'}`}
                    animate={{ 
                        x: anchor.active ? 20 : 0,
                    }}
                    transition={{
                        type: "spring",
                        stiffness: 500,
                        damping: 30
                    }}
                 />
              </div>
            </div>
          </motion.div>
        ))}
      </CardContent>
    </Card>
  )
}

3. Vitality Gauge: "Breathing" Physics
Update components/vitality-gauge.tsx. Using framer-motion allows us to create a perfect sine-wave breathing animation that feels organic, not robotic.
"use client"

import { useState } from "react"
import { Card, CardContent, CardHeader, CardTitle, CardDescription } from "@/components/ui/card"
import { motion, AnimatePresence } from "framer-motion"

type EnergyLevel = "low" | "medium" | "high"

export function VitalityGauge() {
  const [currentEnergy, setCurrentEnergy] = useState<EnergyLevel>("medium")

  // Configuration for the "Breathing" animation based on state
  const breathingVariants = {
    low: { scale: [1, 1.05, 1], opacity: [0.5, 0.8, 0.5] },
    medium: { scale: [1, 1.02, 1], opacity: [0.8, 1, 0.8] },
    high: { scale: [1, 1.1, 1], opacity: [0.9, 1, 0.9] },
  }

  const breathingTransition = {
    low: { duration: 6, repeat: Infinity, ease: "easeInOut" }, // Slow, deep breath
    medium: { duration: 4, repeat: Infinity, ease: "easeInOut" },
    high: { duration: 0.5, repeat: Infinity, ease: "easeInOut" }, // Fast pulse
  }

  const handleCycle = () => {
    setCurrentEnergy(prev => prev === "low" ? "medium" : prev === "medium" ? "high" : "low")
  }

  return (
    <Card className="h-full border border-white/10 bg-black/40 backdrop-blur-xl relative overflow-hidden flex flex-col">
       {/* Background Dynamic Gradient */}
       <motion.div 
         className="absolute inset-0 opacity-20 pointer-events-none"
         animate={{
            background: currentEnergy === 'low' 
                ? 'radial-gradient(circle at 50% 50%, #06b6d4 0%, transparent 70%)' 
                : currentEnergy === 'medium'
                ? 'radial-gradient(circle at 50% 50%, #3b82f6 0%, transparent 70%)'
                : 'radial-gradient(circle at 50% 50%, #a855f7 0%, transparent 70%)'
         }}
         transition={{ duration: 1.5 }}
       />

      <CardHeader className="pb-4 z-10">
         <CardTitle className="text-xl font-medium text-gray-200">Vitality Gauge</CardTitle>
         <CardDescription>System State</CardDescription>
      </CardHeader>
      
      <CardContent className="flex-1 flex flex-col gap-8 z-10 items-center justify-center">
        
        {/* The Orb */}
        <motion.div 
            onClick={handleCycle}
            className="relative cursor-pointer"
            whileHover={{ scale: 1.02 }}
            whileTap={{ scale: 0.95 }}
        >
            {/* The Outer Rings */}
            <div className="w-64 h-64 rounded-full border border-white/5 flex items-center justify-center relative">
                {/* The "Breath" */}
                <motion.div 
                    className={`absolute inset-0 rounded-full blur-2xl ${
                        currentEnergy === 'low' ? 'bg-cyan-500/20' : 
                        currentEnergy === 'medium' ? 'bg-blue-500/20' : 'bg-purple-500/30'
                    }`}
                    variants={breathingVariants}
                    animate={currentEnergy}
                    transition={breathingTransition[currentEnergy]}
                />
                
                {/* The Core Text */}
                <div className="text-center relative z-10">
                    <motion.div 
                        key={currentEnergy} // Triggers animation on change
                        initial={{ y: 20, opacity: 0 }}
                        animate={{ y: 0, opacity: 1 }}
                        exit={{ y: -20, opacity: 0 }}
                        className="text-5xl font-black uppercase tracking-tighter text-white"
                    >
                        {currentEnergy}
                    </motion.div>
                </div>
            </div>
        </motion.div>

        {/* Action Suggestion */}
        <AnimatePresence mode="wait">
            <motion.div 
                key={currentEnergy}
                initial={{ opacity: 0, y: 10 }}
                animate={{ opacity: 1, y: 0 }}
                exit={{ opacity: 0, y: -10 }}
                className="text-center"
            >
                <div className="text-xs text-gray-500 uppercase tracking-widest mb-2">Protocol</div>
                <div className="text-lg font-light text-cyan-100">
                    {currentEnergy === 'low' && "Disconnect. Hydrate. Breathe."}
                    {currentEnergy === 'medium' && "Maintain flow. Clear small tasks."}
                    {currentEnergy === 'high' && "Execute heavy lift. Build Sunroom."}
                </div>
            </motion.div>
        </AnimatePresence>

      </CardContent>
    </Card>
  )
}

4. Brain Dump: "Invisible Secretary"
Update components/brain-dump.tsx. This version introduces the concept of an AI processing state. Items don't just "sit" there; they have a "Process" button that simulates sending them to your other modules.
"use client"

import { useState } from "react"
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card"
import { motion, AnimatePresence } from "framer-motion"

export function BrainDump() {
  const [items, setItems] = useState([
    { id: 1, text: "Buy sanding blocks 120 grit" },
    { id: 2, text: "Call pediatrician for appointment" },
  ])
  const [inputValue, setInputValue] = useState("")
  const [isProcessing, setIsProcessing] = useState(false)

  const addItem = (e: React.FormEvent) => {
    e.preventDefault()
    if (!inputValue.trim()) return
    setItems([{ id: Date.now(), text: inputValue }, ...items])
    setInputValue("")
  }

  const simulateAIProcessing = () => {
    setIsProcessing(true)
    // In a real app, this would call OpenAI
    setTimeout(() => {
        setItems([]) // Clear items as if they were moved
        setIsProcessing(false)
    }, 2000)
  }

  return (
    <Card className="border border-white/10 bg-black/40 backdrop-blur-xl h-full">
      <CardHeader className="flex flex-row items-center justify-between pb-2">
        <CardTitle className="text-xs font-bold tracking-[0.2em] text-gray-500 uppercase">Brain Dump</CardTitle>
        {items.length > 0 && (
            <button 
                onClick={simulateAIProcessing}
                className="text-[10px] uppercase tracking-wider bg-white/10 hover:bg-white/20 px-2 py-1 rounded text-cyan-400 transition-colors"
            >
                {isProcessing ? "Sorting..." : "Process with AI"}
            </button>
        )}
      </CardHeader>
      <CardContent className="space-y-4">
        <form onSubmit={addItem}>
            <input 
                className="w-full bg-white/5 border border-white/10 rounded-lg px-3 py-2 text-sm text-white placeholder:text-gray-600 focus:outline-none focus:border-cyan-500/50 transition-colors"
                placeholder="Unload your mind..."
                value={inputValue}
                onChange={(e) => setInputValue(e.target.value)}
            />
        </form>

        <ul className="space-y-2">
            <AnimatePresence>
                {items.map((item) => (
                    <motion.li
                        key={item.id}
                        initial={{ opacity: 0, height: 0, y: -10 }}
                        animate={{ opacity: 1, height: "auto", y: 0 }}
                        exit={{ opacity: 0, height: 0, x: 20 }} // Slide out on delete
                        className="flex items-center gap-3 p-3 rounded-lg bg-white/5 border border-white/5 group"
                    >
                        <div className="w-1.5 h-1.5 rounded-full bg-cyan-500/50" />
                        <span className="text-sm text-gray-300 flex-1">{item.text}</span>
                    </motion.li>
                ))}
                {items.length === 0 && !isProcessing && (
                    <motion.div 
                        initial={{ opacity: 0 }} 
                        animate={{ opacity: 1 }} 
                        className="text-center text-xs text-gray-700 py-4"
                    >
                        Mind clear.
                    </motion.div>
                )}
            </AnimatePresence>
        </ul>
      </CardContent>
    </Card>
  )
}
